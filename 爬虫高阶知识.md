## 一、AST 级别的 JavaScript 反混淆

### 1.1 为什么需要 AST

正则替换和简单 Hook 能应对初级混淆，但面对**控制流平坦化**、**字符串数组引用**、**花指令**等高级混淆，必须在 AST（抽象语法树）层面做代码还原。

### 1.2 环境搭建

```bash
npm init -y
npm install @babel/parser @babel/traverse @babel/generator @babel/types
```

### 1.3 基本流程

```javascript
const parser = require('@babel/parser');
const traverse = require('@babel/traverse').default;
const generator = require('@babel/generator').default;
const t = require('@babel/types');
const fs = require('fs');

// 1. 读取混淆代码
const code = fs.readFileSync('obfuscated.js', 'utf-8');

// 2. 解析为 AST
const ast = parser.parse(code, {
    sourceType: 'unambiguous',
    // 如果使用 Object.defineProperty 等，取消注释
    // plugins: ['jsx', 'typescript']
});

// 3. 遍历并转换 AST
traverse(ast, {
    // 在这里写各种 Visitor
});

// 4. 生成代码
const output = generator(ast, {
    compact: false,        // 不压缩
    comments: false,       // 去除注释
    retainLines: false,
}).code;

fs.writeFileSync('deobfuscated.js', output);
```

### 1.4 实战：还原字符串数组

混淆特征：`_0x1234('0x1', 'abcd')` → 解码为真实字符串

```javascript
// 目标：找到字符串数组并替换所有引用

traverse(ast, {
    // Step 1: 找到字符串数组定义
    VariableDeclarator(path) {
        const { id, init } = path.node;
        if (
            t.isIdentifier(id) &&
            t.isArrayExpression(init) &&
            init.elements.every(el => t.isStringLiteral(el))
        ) {
            const arrayName = id.name;
            const stringArray = init.elements.map(el => el.value);
            
            // 查找解字符串的函数
            let decodeFuncName = '';
            let decoder = null;
            
            path.parentPath.parentPath.traverse({
                FunctionDeclaration(funcPath) {
                    if (
                        funcPath.node.params.length === 2 &&
                        funcPath.node.body.body.length > 0
                    ) {
                        // 简单判断：函数体包含对数组的引用
                        const funcCode = generator(funcPath.node).code;
                        if (funcCode.includes(arrayName)) {
                            decodeFuncName = funcPath.node.id.name;
                            decoder = funcPath.node;
                        }
                    }
                }
            });
            
            if (decoder && decodeFuncName) {
                // Step 2: 遍历所有调用，计算真实值
                const scope = path.parentPath.scope;
                
                path.parentPath.parentPath.traverse({
                    CallExpression(callPath) {
                        if (
                            t.isIdentifier(callPath.node.callee) &&
                            callPath.node.callee.name === decodeFuncName &&
                            callPath.node.arguments.length === 2 &&
                            callPath.node.arguments.every(
                                arg => t.isStringLiteral(arg) || t.isNumericLiteral(arg)
                            )
                        ) {
                            // 执行解码函数得出真实值
                            // 方法：eval 解码函数 + 调用
                            const funcStr = `(${generator(decoder).code})`;
                            const args = callPath.node.arguments.map(
                                a => JSON.stringify(a.value)
                            ).join(', ');
                            
                            try {
                                const result = eval(`${funcStr}(${args})`);
                                if (typeof result === 'string') {
                                    callPath.replaceWith(
                                        t.stringLiteral(result)
                                    );
                                }
                            } catch(e) {}
                        }
                    }
                });
            }
            
            // 删除数组定义和编码函数
            path.parentPath.remove();
        }
    }
});
```

### 1.5 实战：控制流平坦化还原

混淆特征：`switch(state){ case 0: ... case 1: ... case 2: ... }` 由分发器变量驱动

```javascript
// 还原思路：
// 1. 找到分发器变量和 switch 语句
// 2. 按 case 顺序拼接真实执行流
// 3. 删除分发器

traverse(ast, {
    FunctionDeclaration(path) {
        const body = path.node.body.body;
        
        for (let i = 0; i < body.length; i++) {
            const stmt = body[i];
            
            // 找到 while-switch 模式
            if (
                t.isWhileStatement(stmt) &&
                t.isBlockStatement(stmt.body) &&
                stmt.body.body.length === 1 &&
                t.isSwitchStatement(stmt.body.body[0])
            ) {
                const switchStmt = stmt.body.body[0];
                const discriminator = switchStmt.discriminant;
                
                if (t.isMemberExpression(discriminator)) {
                    // 收集所有 case
                    const cases = {};
                    switchStmt.cases.forEach(c => {
                        if (t.isNumericLiteral(c.test) || t.isStringLiteral(c.test)) {
                            cases[c.test.value] = c.consequent;
                        }
                    });
                    
                    // 尝试按顺序拼接执行流
                    const orderedBody = [];
                    let currentCase = '0'; // 从 case 0 开始
                    const visited = new Set();
                    
                    while (cases[currentCase] && !visited.has(currentCase)) {
                        visited.add(currentCase);
                        const stmts = cases[currentCase];
                        
                        for (const s of stmts) {
                            // 检查是否是分发器更新语句
                            if (
                                t.isExpressionStatement(s) &&
                                t.isAssignmentExpression(s.expression) &&
                                generator(s.expression.left).code === generator(discriminator).code
                            ) {
                                // 找到下一个 case
                                if (t.isNumericLiteral(s.expression.right)) {
                                    currentCase = String(s.expression.right.value);
                                } else if (t.isStringLiteral(s.expression.right)) {
                                    currentCase = s.expression.right.value;
                                }
                            } else {
                                orderedBody.push(s);
                            }
                        }
                        
                        // 如果没有在 case 体中找到下一个值，检查最后一个是否是 continue
                        if (currentCase === undefined) break;
                    }
                    
                    if (orderedBody.length > 0) {
                        // 替换整个 while-switch 为有序代码
                        body.splice(i, 1, ...orderedBody);
                    }
                }
            }
        }
    }
});
```

## 二、WebAssembly（WASM）逆向

越来越多的网站将核心加密逻辑编译成 WASM，直接在 JS 层抄算法已经不够了。

### 2.1 WASM 基础知识

```
C/C++/Rust 源码
     ↓ 编译
  .wasm 字节码 (二进制格式)
     ↓ 浏览器加载
  WebAssembly 实例
     ↓ JS 调用
  wasm.exports.encrypt(data)
```

### 2.2 从浏览器提取 WASM

```javascript
// 方法1：Hook WebAssembly.instantiate
(function() {
    var _instantiate = WebAssembly.instantiate;
    WebAssembly.instantiate = function(bufferSource, importObject) {
        console.log('[+] WebAssembly.instantiate called');
        
        // 保存 wasm 二进制
        if (bufferSource instanceof ArrayBuffer) {
            var blob = new Blob([bufferSource]);
            var url = URL.createObjectURL(blob);
            console.log('[+] WASM Blob URL:', url);
            
            // 在 Console 中可右键链接下载
        }
        
        return _instantiate.call(this, bufferSource, importObject).then(result => {
            console.log('[+] WASM exports:', Object.keys(result.instance.exports));
            return result;
        });
    };
})();
```

```javascript
(async function() {
    // 拦截原始的 instantiate
    const origInstantiate = WebAssembly.instantiate;
    
    WebAssembly.instantiate = async function(bufferSource, importObj) {
        const result = await origInstantiate.call(this, bufferSource, importObj);
        const instance = result.instance;
        
        // Hook 每个导出函数
        for (const [name, func] of Object.entries(instance.exports)) {
            if (typeof func === 'function') {
                const original = func;

                    const result = original.apply(this, args);
                    console.log(`[WASM] ${name} returned:`, result);
                    return result;
                };
            }
        }
        
        return result;
    };
})();
```

## 三、TLS 指纹（JA3/JA4）对抗

### 3.1 什么是 TLS 指纹

TLS 握手时，客户端发送的 **ClientHello** 包含密码套件、扩展列表等信息。这些字段的组合形成指纹，服务端可据此识别非浏览器客户端。

### 3.2 JA3 指纹原理

```
JA3 = MD5(
    SSLVersion,
    CipherSuites (逗号分隔),
    Extensions (逗号分隔，减号分隔),
    EllipticCurves,
    EllipticCurvePointFormats
)
```

Python 的 `requests` 和浏览器的 TLS 指纹不同，CDN/WAF 可以轻易识别。

### 3.3 tls-client — Python 的 TLS 指纹伪装


## 五、CDP（Chrome DevTools Protocol）级别操控

绕过 Playwright/Selenium 的上层封装，直接通过 CDP 协议操控浏览器，实现更精细的控制和更隐蔽的操作。

### 5.1 直接使用 CDP 协议

```python
import asyncio
import json
import websockets

async def cdp_raw_control():
    # 1. 启动 Chrome 带调试端口
    # chrome --remote-debugging-port=9222 --headless=new
    
    # 2. 获取 WebSocket 端点
    import aiohttp
    async with aiohttp.ClientSession() as session:
        async with session.get('http://localhost:9222/json/version') as resp:
            data = await resp.json()
            ws_url = data['webSocketDebuggerUrl']
    
    # 3. 连接并发送 CDP 命令
    async with websockets.connect(ws_url) as ws:
        # Page.navigate
        await ws.send(json.dumps({
            'id': 1,
            'method': 'Page.navigate',
            'params': {'url': 'https://example.com'}
        }))
        response = await ws.recv()
        print(response)
        
        # Runtime.evaluate — 执行 JS
        await ws.send(json.dumps({
            'id': 2,
            'method': 'Runtime.evaluate',
            'params': {
                'expression': 'document.title',
                'returnByValue': True
            }
        }))
        result = await ws.recv()
        print(json.loads(result))
        
        # Network.setRequestInterception — 拦截请求
        await ws.send(json.dumps({
            'id': 3,
            'method': 'Network.enable',
            'params': {}
        }))
        
        await ws.send(json.dumps({
            'id': 4,
            'method': 'Fetch.enable',
            'params': {
                'patterns': [{'urlPattern': '*'}]
            }
        }))

asyncio.run(cdp_raw_control())
```

## 八、代码虚拟化保护（VM Protection）分析

### 8.1 JSVMP 概念

JS 层的 VMP 将原始 JS 代码编译成自定义字节码，用一个解释器循环执行。

```
原始代码: a = b + c
     ↓ 虚拟化
字节码:   [OP_LOAD, 0x01] [OP_LOAD, 0x02] [OP_ADD] [OP_STORE, 0x00]
     ↓ 解释器
while (pc < code.length) {
    switch(code[pc++]) {
        case OP_LOAD: push(var[code[pc++]]); break;
        case OP_ADD:  push(pop() + pop()); break;
        case OP_STORE: var[code[pc++]] = pop(); break;
    }
}
```

### 8.2 识别 VMP

```
1. 代码中存在大数组（字节码）
2. 巨大的 while(true) 或 for(;;) 循环
3. 大量 switch-case 或 if-else 判断
4. 没有明显的业务逻辑函数
```

### 8.3 对抗策略

```
策略1：符号执行
  - 使用 angr、Triton 等符号执行引擎
  - 喂入输入 → 追踪执行路径 → 还原逻辑

策略2：字节码翻译
  - Dump 字节码数组
  - 分析解释器的每个 case 分支
  - 将字节码翻译回可读代码

策略3：RPC 调用（最简单）
  - 不还原逻辑，在浏览器中直接调用虚拟化后的函数
  - 适合逻辑不变、参数可控的场景
```

### 8.4 So 层 VMP（如梆梆、360）

```python
class TaskScheduler:

            'url': url,
            'priority': priority,

        self.producer.send(
            'crawler_tasks',
            value=task,
            key=str(priority).encode(),  # 相同 key 路由到同一分区
        )
    
    def submit_batch(self, tasks):
        """批量提交"""
        for task in tasks:
            self.submit_task(**task)
        self.producer.flush()
```

### 9.4 URL 调度策略

```python
class PriorityURLScheduler:
    """基于优先级的 URL 调度"""
    
    def __init__(self, redis_client):
        self.redis = redis_client
        self.queues = {
            'high': 'url_queue:high',      # 高优：核心业务
            'normal': 'url_queue:normal',  # 普通：常规抓取
            'low': 'url_queue:low',        # 低优：补充抓取
        }
        self.bloom_filter = BloomFilter()
    
    def push(self, url, priority='normal', next_crawl_time=None):
        """添加 URL"""
        if self.bloom_filter.contains(url):
            return False
        
        self.bloom_filter.add(url)
        
        if next_crawl_time:
            # 定时抓取
            self.redis.zadd('url_schedule', {url: next_crawl_time})
        else:
            self.redis.lpush(self.queues[priority], url)
        return True
    
    def pop(self):
        """获取下一个要抓取的 URL（优先级 + 定时）"""
        # 1. 检查是否有到期的定时任务
        now = time.time()
        due = self.redis.zrangebyscore('url_schedule', 0, now, start=0, num=1)
        if due:
            self.redis.zrem('url_schedule', due[0])
            return due[0].decode()
        
        # 2. 按优先级取
        for priority in ['high', 'normal', 'low']:
            url = self.redis.rpop(self.queues[priority])
            if url:
                return url.decode()
        
        return None
```

### 10.2 数据清洗流水线

```python
class DataPipeline:
    """数据管道：清洗 → 去重 → 分类 → 存储"""
    
    def __init__(self):
        self.cleaners = [
            self.strip_html,
            self.normalize_whitespace,
            self.remove_control_chars,
        ]
        self.deduper = BloomFilter(capacity=100_000_000)
        self.classifier = None  # 可接入 ML 模型
    
    async def process(self, raw_item):
        # 1. 清洗
        item = raw_item
        for cleaner in self.cleaners:
            item = cleaner(item)
        
        # 2. 去重
        fingerprint = hashlib.sha256(
            json.dumps(item, sort_keys=True).encode()
        ).hexdigest()
        if self.deduper.contains(fingerprint):
            return None  # 跳过
        self.deduper.add(fingerprint)
        item['_fingerprint'] = fingerprint
        
        # 3. 分类
        if self.classifier:
            item['category'] = self.classifier.predict(item['content'])
        
        # 4. 格式化
        item['processed_at'] = datetime.utcnow().isoformat()
        
        return item
```

```python
import random
import math

class HumanBehaviorSimulator:
    """模拟人类浏览行为"""
    
    @staticmethod
    def human_mouse_move(page, from_x, from_y, to_x, to_y):
        """贝塞尔曲线 + 随机扰动 模拟鼠标移动"""
        steps = random.randint(30, 80)
        cx1 = from_x + (to_x - from_x) * 0.3 + random.randint(-50, 50)
        cy1 = from_y + (to_y - from_y) * random.uniform(0.1, 0.4)
        cx2 = from_x + (to_x - from_x) * 0.7 + random.randint(-30, 30)
        cy2 = from_y + (to_y - from_y) * random.uniform(0.6, 0.9)
        
        for t in range(steps + 1):
            t_norm = t / steps
            # 三次贝塞尔
            x = ((1 - t_norm) ** 3 * from_x +
                 3 * (1 - t_norm) ** 2 * t_norm * cx1 +
                 3 * (1 - t_norm) * t_norm ** 2 * cx2 +
                 t_norm ** 3 * to_x)
            y = ((1 - t_norm) ** 3 * from_y +
                 3 * (1 - t_norm) ** 2 * t_norm * cy1 +
                 3 * (1 - t_norm) * t_norm ** 2 * cy2 +
                 t_norm ** 3 * to_y)
            
            # 添加微噪声
            x += random.gauss(0, 0.5)
            y += random.gauss(0, 0.5)
            
            yield x, y
    
    @staticmethod
    async def human_scroll(page, total_scroll=3000):
        """模拟人类滚动"""
        current = 0
        while current < total_scroll:
            # 不均匀的滚动距离
            delta = random.randint(100, 500)
```
